                                    | '<ImageView' <widgth-height> <src> <padding> <id> <self-closing>

                                    | '<Button' <widgth-height> <text> <padding> <id> <self-closing>

                                    | <num>
<id> ::=                            'android:id=' <alphanum>
<orientation> ::=                   'android:orientation=' <alphanum>

<tag-end> ::=                       '>'
<self-closing> ::=                  '/>'
<alphanum> ::=                      '"'<alphanum-tail> <alphanum> '"'
                                    | '"' <alphanum-tail> <empty> '"'

                                    | '"' <num_tail> <empty> '"'
