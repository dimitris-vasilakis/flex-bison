<xml> ::= <root>
<root> ::= <start-tag> <elements> <end-tag> 
<elements> ::= <element> | <element> <elements> | <empty>
<element> ::= <comment> <start-tag> <content> <end-tag> <comment> | <start-tag> <comment>
<content> ::= <text> | <text> <content> | <element> <content> | <comment> <content> | <empty>
<start-tag> ::= '<' <tag-name> <attributes> ('>' | '/>')
<end-tag> ::= '</' <tag-name> '>'
<comment> ::= <empty> |'<!--' <alphanumeric_tail> '-->'
<attributes> ::= <attribute> | <attribute> <attributes> | <empty>
<attribute> ::= <attribute-name> '=' <attribute_value> 
<tag-name> ::= <alphanumeric> 
<attribute-name> ::= <alphanumeric> 
<text> ::= <alphanumeric_tail>
<attribute_value> ::= '"' <alphanumeric_tail> '"'
<alphanumeric> ::= <letters> <alphanumeric_tail>
<alphanumeric_tail> ::= (<alphanumeric_char> <alphanumeric_tail>) | <empty>
<alphanumeric_char> ::= <letters> | <numbers> | <symbols>
<numbers> ::= <number> | <number> <numbers>
<letters> ::= a|b|...|z|A|B|...|Z
<number> ::= 0|1|...|9
<symbols> ::= ':'|'_'|'-'|';'|'+'|...
